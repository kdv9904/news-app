algorithm check_cycle_or_not(graph): visited = [false] * V // initialize visited array with all false values recursion_stack = [false] * V // initialize recursion stack array with all false values for each vertex v in graph: if not visited[v]: if dfs(v, visited, recursion_stack, -1): return true // cycle found return false // no cycle found function dfs(vertex, visited, recursion_stack, parent): visited[vertex] = true recursion_stack[vertex] = true for each neighbor of vertex: if not visited[neighbor]: if dfs(neighbor, visited, recursion_stack, vertex): return true elif recursion_stack[neighbor] and neighbor != parent: return true // cycle found recursion_stack[vertex] = false return false
